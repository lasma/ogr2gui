37 /* CheckExtensionConsistency() */
38 /* */
39 /* Check that the target file extension is consistant with the */
40 /* requested driver. Actually, we only warn in cases where the */
41 /* inconsistency is blatant (use of an extension declared by one */
42 /* or several drivers, and not by the selected one) */
43 /* -------------------------------------------------------------------- */
44 
45 void CheckExtensionConsistency(const char* pszDestFilename,
46  const char* pszDriverName)
47 {
48 
49  char* pszDestExtension = CPLStrdup(CPLGetExtension(pszDestFilename));
50  if (pszDestExtension[0] != '\0')
51  {
52  int nDriverCount = GDALGetDriverCount();
53  CPLString osConflictingDriverList;
54  for(int i=0;i<nDriverCount;i++)
55  {
56  GDALDriverH hDriver = GDALGetDriver(i);
57  const char* pszDriverExtension =
58  GDALGetMetadataItem( hDriver, GDAL_DMD_EXTENSION, NULL );
59  if (pszDriverExtension && EQUAL(pszDestExtension, pszDriverExtension))
60  {
61  if (GDALGetDriverByName(pszDriverName) != hDriver)
62  {
63  if (osConflictingDriverList.size())
64  osConflictingDriverList += ", ";
65  osConflictingDriverList += GDALGetDriverShortName(hDriver);
66  }
67  else
68  {
69  /* If the request driver allows the used extension, then */
70  /* just stop iterating now */
71  osConflictingDriverList = "";
72  break;
73  }
74  }
75  }
76  if (osConflictingDriverList.size())
77  {
78  fprintf(stderr,
79  "Warning: The target file has a '%s' extension, which is normally used by the %s driver%s,\n"
80  "but the requested output driver is %s. Is it really what you want ?\n",
81  pszDestExtension,
82  osConflictingDriverList.c_str(),
83  strchr(osConflictingDriverList.c_str(), ',') ? "s" : "",
84  pszDriverName);
85  }
86  }
87 
88  CPLFree(pszDestExtension);
89 }

95 void EarlySetConfigOptions( int argc, char ** argv )
96 {
97  /* Must process some config options before GDALAllRegister() or OGRRegisterAll(), */
98  /* but we can't call GDALGeneralCmdLineProcessor() or OGRGeneralCmdLineProcessor(), */
99  /* because it needs the drivers to be registered for the --format or --formats options */
100  for( int i = 1; i < argc; i++ )
101  {
102  if( EQUAL(argv[i],"--config") && i + 2 < argc &&
103  (EQUAL(argv[i + 1], "GDAL_SKIP") ||
104  EQUAL(argv[i + 1], "GDAL_DRIVER_PATH") ||
105  EQUAL(argv[i + 1], "OGR_SKIP") ||
106  EQUAL(argv[i + 1], "OGR_DRIVER_PATH")) )
107  {
108  CPLSetConfigOption( argv[i+1], argv[i+2] );
109 
110  i += 2;
111  }
112  }
113 }
