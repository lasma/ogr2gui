143 void WFSConnect::evtBtnConnect( void )
144 {
145  lstTables->clear();
146  QMessageBox msg;
147  Ogr ogr;
148  QStringList fileList;
149  QString filename = connectionType + txtHost->text();
150  if(ogr.OpenWFS(filename, fileList)) {
151  QStringList::Iterator it = fileList.begin();
152  while(it != fileList.end()) {
153  QListWidgetItem *item = new QListWidgetItem(*it);
154  item->setCheckState( Qt::Unchecked );
155  lstTables->addItem( item );
156  ++it;
157  }
158  btnAccept->setEnabled( true );
159  } else {
160  msg.setText( "* can't connect to web service !" );
161  msg.exec();
162  }
163 }

181 void WFSConnect::evtBtnAccept( void )
182 {
183  connectionString = txtHost->text();
184  selectedLayers.clear();
185  selectedLayersList.clear();
186  for(int i = 0; i < lstTables->count(); ++i) {

188  selectedLayers += lstTables->item(i)->text() + tr(" ");
