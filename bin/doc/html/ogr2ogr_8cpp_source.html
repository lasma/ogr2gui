232  if (poDS == NULL)
233  return NULL;
234 
235  if (pszSQL != NULL)
236  poLyr = poDS->ExecuteSQL( pszSQL, NULL, NULL );
237  else if (pszLyr != NULL)
238  poLyr = poDS->GetLayerByName(pszLyr);
239  else
240  poLyr = poDS->GetLayer(0);
241 
242  if (poLyr == NULL)

245  OGRDataSource::DestroyDataSource(poDS);

247  }

250  poLyr->SetAttributeFilter(pszWhere);
251

253  {
254  OGRGeometry* poSrcGeom = poFeat->GetGeometryRef();
255  if (poSrcGeom)
256  {
257  OGRwkbGeometryType eType = wkbFlatten( poSrcGeom->getGeometryType() );
258

260  poGeom = OGRGeometryFactory::createGeometry( wkbMultiPolygon );
261

263  ((OGRGeometryCollection*)poGeom)->addGeometry( poSrcGeom );
264  else if( eType == wkbMultiPolygon )
265  {

268 
269  for( iGeom = 0; iGeom < nGeomCount; iGeom++ )
270  {

272  ((OGRGeometryCollection*)poSrcGeom)->getGeometryRef(iGeom) );

274  }

280  if( pszSQL != NULL )
281  poDS->ReleaseResultSet( poLyr );
282  OGRDataSource::DestroyDataSource(poDS);
283  return NULL;
284  }
285  }
286 
287  OGRFeature::DestroyFeature(poFeat);
288  }
289 
290  if( pszSQL != NULL )
291  poDS->ReleaseResultSet( poLyr );
292  OGRDataSource::DestroyDataSource(poDS);
293 
294  return poGeom;

415 
416 int OGRSplitListFieldLayer::BuildLayerDefn(GDALProgressFunc pfnProgress,
417  void *pProgressArg)
418 {
419  CPLAssert(poFeatureDefn == NULL);
420 
421  OGRFeatureDefn* poSrcFieldDefn = poSrcLayer->GetLayerDefn();
422 
423  int nSrcFields = poSrcFieldDefn->GetFieldCount();
424  pasListFields =
425  (ListFieldDesc*)CPLCalloc(sizeof(ListFieldDesc), nSrcFields);
426  nListFieldCount = 0;
427  int i;
428 
429  /* Establish the list of fields of list type */
430  for(i=0;i<nSrcFields;i++)
431  {
432  OGRFieldType eType = poSrcFieldDefn->GetFieldDefn(i)->GetType();
433  if (eType == OFTIntegerList ||
434  eType == OFTRealList ||
435  eType == OFTStringList)
436  {
437  pasListFields[nListFieldCount].iSrcIndex = i;
438  pasListFields[nListFieldCount].eType = eType;
439  if (nMaxSplitListSubFields == 1)
440  pasListFields[nListFieldCount].nMaxOccurences = 1;
441  nListFieldCount++;
442  }
443  }
444 
445  if (nListFieldCount == 0)
446  return FALSE;
447 
448  /* No need for full scan if the limit is 1. We just to have to create */
449  /* one and a single one field */
450  if (nMaxSplitListSubFields != 1)
451  {
452  poSrcLayer->ResetReading();
453  OGRFeature* poSrcFeature;
454 
455  int nFeatureCount = 0;
456  if (poSrcLayer->TestCapability(OLCFastFeatureCount))
457  nFeatureCount = poSrcLayer->GetFeatureCount();
458  int nFeatureIndex = 0;
459 
460  /* Scan the whole layer to compute the maximum number of */
461  /* items for each field of list type */
462  while( (poSrcFeature = poSrcLayer->GetNextFeature()) != NULL )
463  {
464  for(i=0;i<nListFieldCount;i++)
465  {
466  int nCount = 0;
467  OGRField* psField =
468  poSrcFeature->GetRawFieldRef(pasListFields[i].iSrcIndex);
469  switch(pasListFields[i].eType)
470  {
471  case OFTIntegerList:
472  nCount = psField->IntegerList.nCount;
473  break;
474  case OFTRealList:
475  nCount = psField->RealList.nCount;
476  break;
477  case OFTStringList:
478  {
479  nCount = psField->StringList.nCount;
480  char** paList = psField->StringList.paList;
481  int j;
482  for(j=0;j<nCount;j++)
483  {
484  int nWidth = strlen(paList[j]);
485  if (nWidth > pasListFields[i].nWidth)
486  pasListFields[i].nWidth = nWidth;
487  }
488  break;
489  }
490  default:
491  CPLAssert(0);
492  break;
493  }
494  if (nCount > pasListFields[i].nMaxOccurences)
495  {
496  if (nCount > nMaxSplitListSubFields)
497  nCount = nMaxSplitListSubFields;
498  pasListFields[i].nMaxOccurences = nCount;
499  }
500  }
501  OGRFeature::DestroyFeature(poSrcFeature);
502 
503  nFeatureIndex ++;
504  if (pfnProgress != NULL && nFeatureCount != 0)
505  pfnProgress(nFeatureIndex * 1.0 / nFeatureCount, "", pProgressArg);
506  }
507  }
508 
509  /* Now let's build the target feature definition */
510 
511  poFeatureDefn =
512  OGRFeatureDefn::CreateFeatureDefn( poSrcFieldDefn->GetName() );
513  poFeatureDefn->Reference();
514  poFeatureDefn->SetGeomType( wkbNone );
515 
516  for(i=0;i<poSrcFieldDefn->GetGeomFieldCount();i++)
517  {
518  poFeatureDefn->AddGeomFieldDefn(poSrcFieldDefn->GetGeomFieldDefn(i));
519  }
520 
521  int iListField = 0;
522  for(i=0;i<nSrcFields;i++)
523  {
524  OGRFieldType eType = poSrcFieldDefn->GetFieldDefn(i)->GetType();
525  if (eType == OFTIntegerList ||
526  eType == OFTRealList ||
527  eType == OFTStringList)
528  {
529  int nMaxOccurences = pasListFields[iListField].nMaxOccurences;
530  int nWidth = pasListFields[iListField].nWidth;
531  iListField ++;
532  int j;
533  if (nMaxOccurences == 1)
534  {
535  OGRFieldDefn oFieldDefn(poSrcFieldDefn->GetFieldDefn(i)->GetNameRef(),
536  (eType == OFTIntegerList) ? OFTInteger :
537  (eType == OFTRealList) ? OFTReal :
538  OFTString);
539  poFeatureDefn->AddFieldDefn(&oFieldDefn);
540  }
541  else
542  {
543  for(j=0;j<nMaxOccurences;j++)
544  {
545  CPLString osFieldName;
546  osFieldName.Printf("%s%d",
547  poSrcFieldDefn->GetFieldDefn(i)->GetNameRef(), j+1);
548  OGRFieldDefn oFieldDefn(osFieldName.c_str(),
549  (eType == OFTIntegerList) ? OFTInteger :
550  (eType == OFTRealList) ? OFTReal :
551  OFTString);
552  oFieldDefn.SetWidth(nWidth);
553  poFeatureDefn->AddFieldDefn(&oFieldDefn);
554  }
555  }
556  }
557  else
558  {
559  poFeatureDefn->AddFieldDefn(poSrcFieldDefn->GetFieldDefn(i));
560  }
561  }
562 
563  return TRUE;
564 }

571 OGRFeature *OGRSplitListFieldLayer::TranslateFeature(OGRFeature* poSrcFeature)
572 {
573  if (poSrcFeature == NULL)
574  return NULL;
575  if (poFeatureDefn == NULL)
576  return poSrcFeature;
577 
578  OGRFeature* poFeature = OGRFeature::CreateFeature(poFeatureDefn);
579  poFeature->SetFID(poSrcFeature->GetFID());
580  for(int i=0;i<poFeature->GetGeomFieldCount();i++)
581  {
582  poFeature->SetGeomFieldDirectly(i, poSrcFeature->StealGeometry(i));
583  }
584  poFeature->SetStyleString(poFeature->GetStyleString());
585 
586  OGRFeatureDefn* poSrcFieldDefn = poSrcLayer->GetLayerDefn();
587  int nSrcFields = poSrcFeature->GetFieldCount();
588  int iSrcField;
589  int iDstField = 0;
590  int iListField = 0;
591  int j;
592  for(iSrcField=0;iSrcField<nSrcFields;iSrcField++)
593  {
594  OGRFieldType eType = poSrcFieldDefn->GetFieldDefn(iSrcField)->GetType();
595  OGRField* psField = poSrcFeature->GetRawFieldRef(iSrcField);
596  switch(eType)
597  {
598  case OFTIntegerList:
599  {
600  int nCount = psField->IntegerList.nCount;
601  if (nCount > nMaxSplitListSubFields)
602  nCount = nMaxSplitListSubFields;
603  int* paList = psField->IntegerList.paList;
604  for(j=0;j<nCount;j++)
605  poFeature->SetField(iDstField + j, paList[j]);
606  iDstField += pasListFields[iListField].nMaxOccurences;
607  iListField++;
608  break;
609  }
610  case OFTRealList:
611  {
612  int nCount = psField->RealList.nCount;
613  if (nCount > nMaxSplitListSubFields)
614  nCount = nMaxSplitListSubFields;
615  double* paList = psField->RealList.paList;
616  for(j=0;j<nCount;j++)
617  poFeature->SetField(iDstField + j, paList[j]);
618  iDstField += pasListFields[iListField].nMaxOccurences;
619  iListField++;
620  break;
621  }
622  case OFTStringList:
623  {
624  int nCount = psField->StringList.nCount;
625  if (nCount > nMaxSplitListSubFields)
626  nCount = nMaxSplitListSubFields;
627  char** paList = psField->StringList.paList;
628  for(j=0;j<nCount;j++)
629  poFeature->SetField(iDstField + j, paList[j]);
630  iDstField += pasListFields[iListField].nMaxOccurences;
631  iListField++;
632  break;
633  }
634  default:
635  poFeature->SetField(iDstField, psField);
636  iDstField ++;
637  break;
638  }
639  }
640 
641  OGRFeature::DestroyFeature(poSrcFeature);
642 
643  return poFeature;
644 }
